// Neopixel rainbow cycle 
// This code interfaces with a single neopixel (WS2812B) to slowly cycle through the colors of the rainbow.

#include <Adafruit_NeoPixel.h>
#define PIN 6
#define NUMPIXELS 1

// Parameter 1 = number of pixels in strip
// Parameter 2 = Arduino pin number (most are valid)
// Parameter 3 = pixel type flags, add together as needed:
//   NEO_KHZ800  800 KHz bitstream (most NeoPixel products w/WS2812 LEDs)
//   NEO_KHZ400  400 KHz (classic 'v1' (not v2) FLORA pixels, WS2811 drivers)
//   NEO_GRB     Pixels are wired for GRB bitstream (most NeoPixel products)
//   NEO_RGB     Pixels are wired for RGB bitstream (v1 FLORA pixels, not v2)
Adafruit_NeoPixel strip = Adafruit_NeoPixel(NUMPIXELS, PIN, NEO_GRB + NEO_KHZ800);

// IMPORTANT: To reduce NeoPixel burnout risk, add 1000 uF capacitor across
// pixel power leads, add 300 - 500 Ohm resistor on first pixel's data input
// and minimize distance between Arduino and first pixel.  Avoid connecting
// on a live circuit...if you must, connect GND first.

void setup() { 
  strip.begin();
  strip.show(); // Initialize all pixels to 'off'
}

void loop() {
  // Cycle through the rainbow!
  colorWipe(strip.Color(255, 0, 0), 50);   // Red
  colorWipe(strip.Color(255, 165, 0), 50); // Orange
  colorWipe(strip.Color(255, 255, 0), 50); // Yellow
  colorWipe(strip.Color(0, 255, 0), 50);   // Green
  colorWipe(strip.Color(0, 0, 255), 50);   // Blue
  colorWipe(strip.Color(75, 0, 130), 50);  // Indigo 
  colorWipe(strip.Color(148, 0, 211), 50); // Violet
}

// Fill the dots one after the other with a color
void colorWipe(uint32_t c, uint8_t wait) {
  for(uint16_t i=0; i<strip.numPixels(); i++) {
      strip.setPixelColor(i, c);
      strip.show();
      delay(wait);
  }
}
```

// LED 1 on Digital I/O 5

// Neopixel rainbow cycle 
// This code interfaces with a single neopixel (WS2812B) to slowly cycle through the colors of the rainbow.

#include <Adafruit_NeoPixel.h>
#define PIN 6
#define NUMPIXELS 1

// Parameter 1 = number of pixels in strip
// Parameter 2 = Arduino pin number (most are valid)
// Parameter 3 = pixel type flags, add together as needed:
//   NEO_KHZ800  800 KHz bitstream (most NeoPixel products w/WS2812 LEDs)
//   NEO_KHZ400  400 KHz (classic 'v1' (not v2) FLORA pixels, WS2811 drivers)
//   NEO_GRB     Pixels are wired for GRB bitstream (most NeoPixel products)
//   NEO_RGB     Pixels are wired for RGB bitstream (v1 FLORA pixels, not v2)
Adafruit_NeoPixel strip = Adafruit_NeoPixel(NUMPIXELS, PIN, NEO_GRB + NEO_KHZ800);

// IMPORTANT: To reduce NeoPixel burnout risk, add 1000 uF capacitor across
// pixel power leads, add 300 - 500 Ohm resistor on first pixel's data input
// and minimize distance between Arduino and first pixel.  Avoid connecting
// on a live circuit...if you must, connect GND first.

void setup() { 
  strip.begin();
  strip.show(); // Initialize all pixels to 'off'
}

void loop() {
  // Cycle through the rainbow!
  colorWipe(strip.Color(255, 0, 0), 50);   // Red
  colorWipe(strip.Color(255, 165, 0), 50); // Orange
  colorWipe(strip.Color(255, 255, 0), 50); // Yellow
  colorWipe(strip.Color(0, 255, 0), 50);   // Green
  colorWipe(strip.Color(0, 0, 255), 50);   // Blue
  colorWipe(strip.Color(75, 0, 130), 50);  // Indigo 
  colorWipe(strip.Color(148, 0, 211), 50); // Violet
}

// Fill the dots one after the other with a color
void colorWipe(uint32_t c, uint8_t wait) {
  for(uint16_t i=0; i<strip.numPixels(); i++) {
      strip.setPixelColor(i, c);
      strip.show();
      delay(wait);
  }
}
```
