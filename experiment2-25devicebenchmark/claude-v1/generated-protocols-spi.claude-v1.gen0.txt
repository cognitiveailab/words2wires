// SPI Communication
// This code interfaces with another device over SPI. When the string "hello" is received, 
// it responds by sending "world".

#include <SPI.h>

// Slave Select pin
#define SS 10
// Define SPI pins
#define MOSI 11
#define MISO 12
#define SCK 13

// Buffer to hold incoming and outgoing data
char incoming[6]; 
char outgoing[6] = "world";

void setup() {
  // Set SS, MOSI, MISO and SCK as outputs
  pinMode(SS, OUTPUT);
  pinMode(MOSI, OUTPUT);
  pinMode(MISO, INPUT);
  pinMode(SCK, OUTPUT);
  
  // Enable SPI
  SPI.begin();
  
  // Set SS high to deselect slave
  digitalWrite(SS, HIGH);
}

void loop() {
  // Check if data is available
  if (SPI.available()) {
    // Read incoming data
    for (int i = 0; i < 6; i++) {
      incoming[i] = SPI.transfer(0);
    }
    
    // Check if "hello" was received
    if (strcmp(incoming, "hello") == 0) {
      // Select slave
      digitalWrite(SS, LOW);
      
      // Send "world"
      for (int i = 0; i < 6; i++) {
        SPI.transfer(outgoing[i]);
      }
      
      // Deselect slave
      digitalWrite(SS, HIGH);
    }
  }
}
```

// SPI Communication
// This code interfaces with another device over SPI. When the string "hello" is received, 
// it responds by sending "world".

#include <SPI.h>

// Slave Select pin
#define SS 10
// Define SPI pins
#define MOSI 11
#define MISO 12
#define SCK 13

// Buffer to hold incoming and outgoing data
char incoming[6]; 
char outgoing[6] = "world";

void setup() {
  // Set SS, MOSI, MISO and SCK as outputs
  pinMode(SS, OUTPUT);
  pinMode(MOSI, OUTPUT);
  pinMode(MISO, INPUT);
  pinMode(SCK, OUTPUT);
  
  // Enable SPI
  SPI.begin();
  
  // Set SS high to deselect slave
  digitalWrite(SS, HIGH);
}

void loop() {
  // Check if data is available
  if (SPI.available()) {
    // Read incoming data
    for (int i = 0; i < 6; i++) {
      incoming[i] = SPI.transfer(0);
    }
    
    // Check if "hello" was received
    if (strcmp(incoming, "hello") == 0) {
      // Select slave
      digitalWrite(SS, LOW);
      
      // Send "world"
      for (int i = 0; i < 6; i++) {
        SPI.transfer(outgoing[i]);
      }
      
      // Deselect slave
      digitalWrite(SS, HIGH);
    }
  }
}
