// LED on Digital I/O 0
#define PIN_LED 0

// Button pins
const int buttonPins[] = {1, 2, 3, 4, 5,6, 7, 8, 9};
const int numButtons = sizeof(buttonPins) / sizeof(buttonPins[0]);

// ASCII codes for emojis
const char* emojis[] = {
    "â¤ï¸", "ğŸ˜€", "ğŸ˜‚", "ğŸ˜Š", "ğŸ˜", "ğŸ˜", "ğŸ˜¢", "ğŸ˜¡", "ğŸ‘ğŸ‘ğŸ‘ğŸ‘ğŸ‘"
};

// Bounce objects for button debouncing
Bounce buttons[numButtons];

void setup() {
    // Initialize LED pin to output mode
    pinMode(PIN_LED, OUTPUT);
    digitalWrite(PIN_LED, HIGH); // LED on

    // Initialize button pins to input mode with internal pullup
    for (int i = 0; i < numButtons; i++) {
        pinMode(buttonPins[i], INPUT_PULLUP);
        buttons[i].attach(buttonPins[i]);
        buttons[i].interval(5);
    }

    // Start USB keyboard
    Keyboard.begin();
}

void loop() {
    for (int i = 0; i < numButtons; i++) {
        buttons[i].update();
        if (buttons[i].fell()) {
            digitalWrite(PIN_LED, LOW); // LED off
            delay(500);                 // wait for 500ms
            digitalWrite(PIN_LED, HIGH); // LED on
            Keyboard.print(emojis[i]);
        }
    }
}

#define PIN_LED 0

// Button pins
const int buttonPins[] = {1, 2, 3, 4, 5,6, 7, 8, 9};
const int numButtons = sizeof(buttonPins) / sizeof(buttonPins[0]);

// ASCII codes for emojis

    "â¤ï¸", "ğŸ˜€", "ğŸ˜‚", "ğŸ˜Š", "ğŸ˜", "ğŸ˜", "ğŸ˜¢", "ğŸ˜¡", "ğŸ‘ğŸ‘ğŸ‘ğŸ‘ğŸ‘"

// Bounce objects for button debouncing
Bounce buttons[numButtons];

void setup() {
    // Initialize LED pin to output mode
    pinMode(PIN_LED, OUTPUT);
    digitalWrite(PIN_LED, HIGH); // LED on

    // Initialize button pins to input mode with internal pullup
    for (int i = 0; i < numButtons; i++) {
        pinMode(buttonPins[i], INPUT_PULLUP);
        buttons[i].attach(buttonPins[i]);
        buttons[i].interval(5);
    }

    // Start USB keyboard
    Keyboard.begin();
}

void loop() {
    for (int i = 0; i < numButtons; i++) {
        buttons[i].update();
        if (buttons[i].fell()) {
            digitalWrite(PIN_LED, LOW); // LED off
            delay(500);                 // wait for 500ms
            digitalWrite(PIN_LED, HIGH); // LED on
            Keyboard.print(emojis[i]);
        }
    }
}
